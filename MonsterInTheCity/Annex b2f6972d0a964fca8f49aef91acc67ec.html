Here is the vex snippet for creating the points in the modules placer.

float line_length = prim(2,'length',0);
int pieces_amount = nprimitives(1);
//printf(itoa(pieces_amount)+'\n');

float added_length = 0;
int i = 0;
int iteration_id = detail(3,'iteration');
vector source_position = point(2,'P',0);
//vector source_position = {0,0,0};
vector normal = point(2,'N',0);

vector direction = normalize(point(2,"P",1) - point(2,"P",0));
//vector direction = {1,0,0}*detail(2,'direction',0);

//v@direction = direction;
do {
//if line is too short take smallest module
if (detail(1,'min_width',0) > line_length){
int id = addpoint(0,source_position);
setpointattrib(0,'variant',id,string(detail(1,'smallest_piece',0)));
setpointattrib(0,'N',id,normal);
added_length = detail(1,'min_width',0);
}
else{
//places multiple modules
vector position = (direction * added_length) + source_position;

int index = int(fit01(nrandom(),0,pieces_amount));

int id = addpoint(0,position);
setpointattrib(0,'variant',id,string(prim(1,'variant',index)));
setpointattrib(0,'N',id,normal);

added_length += prim(1,'size_x',index);
i += 1;

//prevent infinite loops
if (i > 100){break;}
}
}
//when the module length is over the line length we stop
while(added_length < line_length);
//finnaly we add a last point that will only be used for scaling
vector position = (direction * added_length) + source_position;
int id = addpoint(0,position);

Here is the vex snippet for finding concave and convex points

int points[] = primpoints(0, @primnum);
int num = len(points);
vector up = prim_normal(0, @primnum, 0.5, 0.5);

for(int i=0; i<num; i++) {
    vector pos      = point(0, "P", points[i]);
    vector prev_pos = point(0, "P", points[(i - 1) % num]);    
    vector next_pos = point(0, "P", points[(i + 1) % num]);  
    
    // get vectors pointing to the prev and next point in the prim
    vector edge_prev = normalize(pos - prev_pos);
    vector edge_next = normalize(next_pos - pos);
    
    // gives any concave points concave=1 attrib
    float dist = dot(lerp(next_pos, pos, 0.5) - prev_pos, normalize(cross(edge_next, up)));
    if(dist > 0.5){
        setpointattrib(geoself(), "variant", points[i], 'closed','set');

        }
    else if (dist < -0.5){
        setpointattrib(geoself(), "variant", points[i], 'open','set');
        
        }

}
